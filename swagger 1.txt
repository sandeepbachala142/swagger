swagger: '2.0'
info:
  title: My API
  version: 1.0.0
schemes:
  - https
basePath: /v1
paths:
  /getAccess:
    get:
      summary: Get Access
      description: Retrieves access information
      produces:
        - application/json
      responses:
        200:
          description: Access information retrieved successfully
          schema:
            $ref: '#/definitions/AccessInformation'
  /refresh:
    post:
      summary: Refresh Access
      description: Refreshes the access information
      produces:
        - application/json
      parameters:
        - in: body
          name: refresh_request
          description: The request to refresh access
          required: true
          schema:
            $ref: '#/definitions/RefreshRequest'
      responses:
        200:
          description: Access refreshed successfully
          schema:
            $ref: '#/definitions/AccessInformation'
  /access:
    post:
      summary: Access
      description: Gets access to the system
      produces:
        - application/json
      parameters:
        - in: body
          name: access_request
          description: The request for system access
          required: true
          schema:
            $ref: '#/definitions/AccessRequest'
      responses:
        200:
          description: Access granted successfully
          schema:
            $ref: '#/definitions/AccessToken'

definitions:
  AccessInformation:
    type: object
    properties:
      accessKey:
        type: string
      secretKey:
        type: string

  RefreshRequest:
    type: object
    properties:
      accessKey:
        type: string
      secretKey:
        type: string

  AccessRequest:
    type: object
    properties:
      username:
        type: string
      password:
        type: string

  AccessToken:
    type: object
    properties:
      token:
        type: string
